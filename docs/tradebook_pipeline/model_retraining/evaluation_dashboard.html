522    def _calculate_health_score(self, metrics: Any) -> float:
523        """Calculate overall system health score (0-1)"""
524        scores = []
525        
526        # Accuracy score
527        if metrics.prediction_accuracy >= 0.7:
528            scores.append(1.0)
529        elif metrics.prediction_accuracy >= 0.6:
530            scores.append(0.7)
531        else:
532            scores.append(0.4)
533        
534        # Win rate score
535        if metrics.win_rate >= 0.6:
536            scores.append(1.0)
537        elif metrics.win_rate >= 0.5:
538            scores.append(0.8)
539        else:
540            scores.append(0.5)
541        
542        # Calibration score
543        scores.append(metrics.model_confidence_calibration)
544        
545        # Execution quality (latency)
546        if metrics.avg_execution_latency_ms < 100:
547            scores.append(1.0)
548        elif metrics.avg_execution_latency_ms < 500:
549            scores.append(0.8)
550        else:
551            scores.append(0.6)
552        
553        # Slippage score
554        if metrics.avg_slippage_pct < 0.005:
555            scores.append(1.0)
556        elif metrics.avg_slippage_pct < 0.02:
557            scores.append(0.7)
558        else:
559            scores.append(0.5)
560        
561        return np.mean(scores)

523    def _calculate_health_score(self, metrics: Any) -> float:
524        """Calculate overall system health score (0-1)"""
525        scores = []
526        
527        # Accuracy score
528        if metrics.prediction_accuracy >= 0.7:
529            scores.append(1.0)
530        elif metrics.prediction_accuracy >= 0.6:
531            scores.append(0.7)
532        else:
533            scores.append(0.4)
534        
535        # Win rate score
536        if metrics.win_rate >= 0.6:
537            scores.append(1.0)
538        elif metrics.win_rate >= 0.5:
539            scores.append(0.8)
540        else:
541            scores.append(0.5)
542        
543        # Calibration score
544        scores.append(metrics.model_confidence_calibration)
545        
546        # Execution quality (latency)
547        if metrics.avg_execution_latency_ms < 100:
548            scores.append(1.0)
549        elif metrics.avg_execution_latency_ms < 500:
550            scores.append(0.8)
551        else:
552            scores.append(0.6)
553        
554        # Slippage score
555        if metrics.avg_slippage_pct < 0.005:
556            scores.append(1.0)
557        elif metrics.avg_slippage_pct < 0.02:
558            scores.append(0.7)
559        else:
560            scores.append(0.5)
561        
562        return np.mean(scores)
