 78    def save_new_model_version(self, models: List[Dict[str, Any]], 
 79                               training_samples: int,
 80                               performance_metrics: Dict[str, float]) -> List[str]:
 81        """
 82        Save newly trained models as versioned artifacts.
 83        
 84        Args:
 85            models: List of trained model dictionaries
 86            training_samples: Number of samples used for training
 87            performance_metrics: Performance metrics from training
 88            
 89        Returns:
 90            List of version IDs for saved models
 91        """
 92        with self._lock:
 93            version_ids = []
 94            
 95            for model_dict in models:
 96                try:
 97                    version_id = self._generate_version_id(model_dict)
 98                    model_name = model_dict.get('name', 'unnamed_model')
 99                    
100                    logger.info(f"Saving model version: {model_name} (v{version_id})")
101                    
102                    # Create version directory
103                    version_dir = self.versions_dir / version_id
104                    version_dir.mkdir(parents=True, exist_ok=True)
105                    
106                    # Save model using multiple strategies
107                    model_file = self._save_model_artifact(model_dict, version_dir, version_id)
108                    
109                    if not model_file:
110                        logger.error(f"Failed to save model {model_name}")
111                        continue
112                    
113                    # Create version metadata
114                    version = ModelVersion(
115                        version_id=version_id,
116                        timestamp=datetime.now(),
117                        model_name=model_name,
118                        model_type=model_dict.get('strategy', 'unknown'),
119                        training_samples=training_samples,
120                        performance_metrics=performance_metrics,
121                        file_path=str(model_file),
122                        status='pending',
123                        deployment_strategy=self.deployment_strategy
124                    )
125                    
126                    # Save metadata
127                    metadata_file = version_dir / "metadata.json"
128                    with open(metadata_file, 'w') as f:
129                        json.dump(asdict(version), f, indent=2, default=str)
130                    
131                    # Save model configuration
132                    config_file = version_dir / "config.json"
133                    with open(config_file, 'w') as f:
134                        json.dump({
135                            'name': model_dict.get('name'),
136                            'strategy': model_dict.get('strategy'),
137                            'parameters': model_dict.get('parameters', {}),
138                            'features': model_dict.get('features', [])
139                        }, f, indent=2, default=str)
140                    
141                    # Register version
142                    self.model_registry[version_id] = version
143                    version_ids.append(version_id)
144                    
145                    logger.info(f" Saved model version {version_id}")
146                    
147                except Exception as e:
148                    logger.error(f"Failed to save model {model_dict.get('name', 'unnamed')}: {e}")
149            
150            # Save updated registry
151            self._save_registry()
152            
153            return version_ids

155    def _save_model_artifact(self, model_dict: Dict[str, Any], 
156                            version_dir: Path, version_id: str) -> Optional[Path]:
157        """Save model artifact using best available method"""
158        model = model_dict.get('model')
159        
160        if not model:
161            return None
162        
163        # Try pickle first
164        try:
165            model_file = version_dir / f"{version_id}.pkl"
166            with open(model_file, 'wb') as f:
167                pickle.dump(model, f)
168            logger.debug(f"Saved with pickle: {model_file}")
169            return model_file
170        except Exception as e:
171            logger.debug(f"Pickle save failed: {e}")
172        
173        # Try dill
174        try:
175            model_file = version_dir / f"{version_id}.dill"
176            with open(model_file, 'wb') as f:
177                dill.dump(model, f)
178            logger.debug(f"Saved with dill: {model_file}")
179            return model_file
180        except Exception as e:
181            logger.debug(f"Dill save failed: {e}")
182        
183        # Fallback: save serializable components
184        try:
185            if isinstance(model, dict):
186                serializable = {}
187                for key, value in model.items():
188                    try:
189                        pickle.dumps(value)
190                        serializable[key] = value
191                    except:
192                        pass
193                
194                model_file = version_dir / f"{version_id}_components.pkl"
195                with open(model_file, 'wb') as f:
196                    pickle.dump(serializable, f)
197                logger.warning(f"Saved partial model components: {model_file}")
198                return model_file
199        except Exception as e:
200            logger.error(f"All save methods failed: {e}")
201        
202        return None

204    def deploy_model(self, version_id: str, force: bool = False) -> bool:
205        """
206        Deploy a model version using configured deployment strategy.
207        
208        Args:
209            version_id: Version ID to deploy
210            force: Skip validation if True
211            
212        Returns:
213            bool: Success status
214        """
215        with self._lock:
216            if version_id not in self.model_registry:
217                logger.error(f"Version {version_id} not found in registry")
218                return False
219            
220            version = self.model_registry[version_id]
221            
222            logger.info(f"Deploying model {version.model_name} v{version_id} "
223                       f"with strategy: {self.deployment_strategy}")
224            
225            # Load model
226            model = self._load_model(version_id)
227            if not model:
228                logger.error(f"Failed to load model {version_id}")
229                return False
230            
231            # Deploy based on strategy
232            if self.deployment_strategy == 'immediate' or force:
233                return self._deploy_immediate(version_id, model, version)
234            
235            elif self.deployment_strategy == 'shadow_mode':
236                return self._deploy_shadow(version_id, model, version)
237            
238            elif self.deployment_strategy == 'a_b_test':
239                return self._deploy_ab_test(version_id, model, version)
240            
241            else:
242                logger.error(f"Unknown deployment strategy: {self.deployment_strategy}")
243                return False

245    def _deploy_immediate(self, version_id: str, model: Any, 
246                         version: ModelVersion) -> bool:
247        """Immediately activate new model"""
248        try:
249            # Archive current active model
250            self._archive_active_models()
251            
252            # Set as active
253            self.active_models[version.model_name] = model
254            version.status = 'active'
255            
256            # Copy to active directory
257            self._copy_to_active(version_id)
258            
259            logger.info(f" Model {version_id} deployed immediately")
260            self._save_registry()
261            return True
262            
263        except Exception as e:
264            logger.error(f"Immediate deployment failed: {e}")
265            return False

267    def _deploy_shadow(self, version_id: str, model: Any, 
268                      version: ModelVersion) -> bool:
269        """Deploy in shadow mode for validation"""
270        try:
271            self.shadow_models[version.model_name] = {
272                'model': model,
273                'version_id': version_id,
274                'predictions': [],
275                'trade_count': 0
276            }
277            
278            version.status = 'shadow'
279            
280            logger.info(f" Model {version_id} deployed in shadow mode "
281                       f"({self.validation_trades} trades for validation)")
282            
283            self._save_registry()
284            return True
285            
286        except Exception as e:
287            logger.error(f"Shadow deployment failed: {e}")
288            return False

303            version.status = 'ab_testing'
304            
305            logger.info(f" Model {version_id} deployed for A/B testing")
306            self._save_registry()
307            return True
308            
309        except Exception as e:
310            logger.error(f"A/B deployment failed: {e}")

313    def validate_shadow_model(self, model_name: str) -> bool:
314        """
315        Validate shadow model performance and promote if successful.
316        
317        Returns:
318            bool: True if model promoted to active
319        """
320        with self._lock:
321            if model_name not in self.shadow_models:
322                return False
323            
324            shadow_info = self.shadow_models[model_name]
325            
326            if shadow_info['trade_count'] < self.validation_trades:
327                logger.debug(f"Shadow model validation: {shadow_info['trade_count']}/{self.validation_trades} trades")
328                return False
329            
330            # Calculate shadow model performance
331            predictions = shadow_info['predictions']
332            correct = sum(1 for p in predictions if p.get('correct', False))
333            accuracy = correct / len(predictions) if predictions else 0
334            
335            # Compare with active model
336            active_model = self.active_models.get(model_name)
337            version_id = shadow_info['version_id']
338            version = self.model_registry[version_id]
339            
340            promote = True
341            
342            if active_model:
343                # Get active model version
344                active_version_id = self._get_active_version_id(model_name)
345                if active_version_id:
346                    active_version = self.model_registry[active_version_id]
347                    active_accuracy = active_version.performance_metrics.get('prediction_accuracy', 0)
348                    
349                    # Check if new model is better
350                    if accuracy < active_accuracy * (1 - self.rollback_threshold):
351                        logger.warning(f"Shadow model underperforming: {accuracy:.3f} vs {active_accuracy:.3f}")
352                        promote = False
353            
354            if promote:
355                logger.info(f" Promoting shadow model {version_id} to active (accuracy: {accuracy:.3f})")
356                
357                # Archive current active
358                self._archive_active_models()
359                
360                # Promote shadow to active
361                self.active_models[model_name] = shadow_info['model']
362                version.status = 'active'
363                version.validation_results = {
364                    'validation_trades': shadow_info['trade_count'],
365                    'validation_accuracy': accuracy,
366                    'promoted_at': datetime.now().isoformat()
367                }
368                
369                # Copy to active directory
370                self._copy_to_active(version_id)
371                
372                # Remove from shadow
373                del self.shadow_models[model_name]
374                
375                self._save_registry()
376                return True
377            else:
378                logger.info(f"Shadow model {version_id} did not pass validation")
379                version.status = 'failed_validation'
380                del self.shadow_models[model_name]
381                self._save_registry()
382                return False

384    def record_shadow_prediction(self, model_name: str, prediction: int, 
385                                 actual_outcome: bool) -> None:
386        """Record prediction from shadow model for validation"""
387        with self._lock:
388            if model_name in self.shadow_models:
389                self.shadow_models[model_name]['predictions'].append({
390                    'prediction': prediction,
391                    'correct': actual_outcome,
392                    'timestamp': datetime.now().isoformat()
393                })
394                self.shadow_models[model_name]['trade_count'] += 1
395                
396                # Check if ready for validation
397                if self.shadow_models[model_name]['trade_count'] >= self.validation_trades:
398                    self.validate_shadow_model(model_name)

400    def rollback_to_version(self, version_id: str) -> bool:
401        """Rollback to a previous model version"""
402        with self._lock:
403            logger.warning(f"Rolling back to version {version_id}")
404            
405            if version_id not in self.model_registry:
406                logger.error(f"Version {version_id} not found")
407                return False
408            
409            version = self.model_registry[version_id]
410            model = self._load_model(version_id)
411            
412            if not model:
413                logger.error(f"Failed to load version {version_id}")
414                return False
415            
416            # Archive current active
417            self._archive_active_models()
418            
419            # Restore old version
420            self.active_models[version.model_name] = model
421            version.status = 'active'
422            
423            self._copy_to_active(version_id)
424            self._save_registry()
425            
426            logger.info(f" Rolled back to version {version_id}")
427            return True

429    def get_active_model(self, model_name: str) -> Optional[Any]:
430        """Get currently active model"""

432            return self.active_models.get(model_name)

434    def get_model_versions(self, model_name: Optional[str] = None) -> List[ModelVersion]:
435        """Get all versions, optionally filtered by model name"""
436        with self._lock:
437            versions = list(self.model_registry.values())
438            
439            if model_name:
440                versions = [v for v in versions if v.model_name == model_name]
441            
442            return sorted(versions, key=lambda v: v.timestamp, reverse=True)

444    def _load_model(self, version_id: str) -> Optional[Any]:
445        """Load model from disk"""
446        try:
447            version = self.model_registry[version_id]
448            file_path = Path(version.file_path)
449            
450            if not file_path.exists():
451                logger.error(f"Model file not found: {file_path}")
452                return None
453            
454            # Try appropriate loader based on file extension
455            if file_path.suffix == '.pkl':
456                with open(file_path, 'rb') as f:
457                    return pickle.load(f)
458            elif file_path.suffix == '.dill':
459                with open(file_path, 'rb') as f:
460                    return dill.load(f)
461            else:
462                with open(file_path, 'rb') as f:
463                    return pickle.load(f)
464                    
465        except Exception as e:
466            logger.error(f"Failed to load model {version_id}: {e}")
467            return None

469    def _generate_version_id(self, model_dict: Dict[str, Any]) -> str:
470        """Generate unique version ID"""
471        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
472        model_name = model_dict.get('name', 'model')
473        
474        # Create hash of model configuration
475        config_str = json.dumps({
476            'name': model_name,
477            'strategy': model_dict.get('strategy'),
478            'parameters': model_dict.get('parameters', {})
479        }, sort_keys=True)
480        
481        config_hash = hashlib.md5(config_str.encode()).hexdigest()[:8]
482        
483        return f"{timestamp}_{config_hash}"

485    def _copy_to_active(self, version_id: str) -> None:
486        """Copy model version to active directory"""
487        try:
488            version = self.model_registry[version_id]
489            src = Path(version.file_path)
490            dst = self.active_dir / src.name
491            
492            shutil.copy2(src, dst)
493            
494            # Also copy metadata
495            src_meta = src.parent / "metadata.json"
496            dst_meta = self.active_dir / "metadata.json"
497            shutil.copy2(src_meta, dst_meta)
498            
499        except Exception as e:
500            logger.error(f"Failed to copy to active directory: {e}")
501    
502    def _archive_active_models(self) -> None:
503        """Archive currently active models"""
504        for model_name in list(self.active_models.keys()):
505            version_id = self._get_active_version_id(model_name)
506            if version_id:
507                self.model_registry[version_id].status = 'archived'
508        
509        self.active_models.clear()
510    
511    def _get_active_version_id(self, model_name: str) -> Optional[str]:
512        """Get version ID of currently active model"""
513        for version_id, version in self.model_registry.items():
514            if version.model_name == model_name and version.status == 'active':
515                return version_id
516        return None
517    
518    def _save_registry(self) -> None:
519        """Save model registry to disk"""
520        try:
521            registry_file = self.models_dir / "model_registry.json"
522            registry_data = {
523                version_id: asdict(version) 
524                for version_id, version in self.model_registry.items()
525            }
526            
527            with open(registry_file, 'w') as f:
528                json.dump(registry_data, f, indent=2, default=str)
529                
530        except Exception as e:
531            logger.error(f"Failed to save registry: {e}")
532    
533    def _load_registry(self) -> None:
534        """Load model registry from disk"""
535        try:
536            registry_file = self.models_dir / "model_registry.json"
537            
538            if registry_file.exists():
539                with open(registry_file, 'r') as f:
540                    registry_data = json.load(f)
541                
542                for version_id, data in registry_data.items():
543                    # Convert timestamp string back to datetime
544                    data['timestamp'] = datetime.fromisoformat(data['timestamp'])
545                    self.model_registry[version_id] = ModelVersion(**data)
546                
547                logger.info(f"Loaded {len(self.model_registry)} model versions from registry")
548                
549        except Exception as e:
550            logger.warning(f"Could not load registry: {e}")

314    def validate_shadow_model(self, model_name: str) -> bool:
315        """
316        Validate shadow model performance and promote if successful.
317        
318        Returns:
319            bool: True if model promoted to active
320        """
321        with self._lock:
322            if model_name not in self.shadow_models:
323                return False
324            
325            shadow_info = self.shadow_models[model_name]
326            
327            if shadow_info['trade_count'] < self.validation_trades:
328                logger.debug(f"Shadow model validation: {shadow_info['trade_count']}/{self.validation_trades} trades")
329                return False
330            
331            # Calculate shadow model performance
332            predictions = shadow_info['predictions']
333            correct = sum(1 for p in predictions if p.get('correct', False))
334            accuracy = correct / len(predictions) if predictions else 0
335            
336            # Compare with active model
337            active_model = self.active_models.get(model_name)
338            version_id = shadow_info['version_id']
339            version = self.model_registry[version_id]
340            
341            promote = True
342            
343            if active_model:
344                # Get active model version
345                active_version_id = self._get_active_version_id(model_name)
346                if active_version_id:
347                    active_version = self.model_registry[active_version_id]
348                    active_accuracy = active_version.performance_metrics.get('prediction_accuracy', 0)
349                    
350                    # Check if new model is better
351                    if accuracy < active_accuracy * (1 - self.rollback_threshold):
352                        logger.warning(f"Shadow model underperforming: {accuracy:.3f} vs {active_accuracy:.3f}")
353                        promote = False
354            
355            if promote:
356                logger.info(f" Promoting shadow model {version_id} to active (accuracy: {accuracy:.3f})")
357                
358                # Archive current active
359                self._archive_active_models()
360                
361                # Promote shadow to active
362                self.active_models[model_name] = shadow_info['model']
363                version.status = 'active'
364                version.validation_results = {
365                    'validation_trades': shadow_info['trade_count'],
366                    'validation_accuracy': accuracy,
367                    'promoted_at': datetime.now().isoformat()
368                }
369                
370                # Copy to active directory
371                self._copy_to_active(version_id)
372                
373                # Remove from shadow
374                del self.shadow_models[model_name]
375                
376                self._save_registry()
377                return True
378            else:
379                logger.info(f"Shadow model {version_id} did not pass validation")
380                version.status = 'failed_validation'
381                del self.shadow_models[model_name]
382                self._save_registry()
383                return False
